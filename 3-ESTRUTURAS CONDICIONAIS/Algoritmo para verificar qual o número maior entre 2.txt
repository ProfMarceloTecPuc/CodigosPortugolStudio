//Algoritmo para verificar qual o número maior entre 2

	funcao inicio() {
		inteiro numero1, numero2
		escreva("Digite o número 1: ")
		leia(numero1)
		escreva("Digite o número 2: ")
		leia(numero2)
		se (numero1 > numero2){
    			escreva("Número 1 maior")
		} senao { 
			se (numero1 < numero2){
    				escreva ("Número 2 maior")
			} senao {
   				escreva("Números iguais")
			}
		}
	}
}
